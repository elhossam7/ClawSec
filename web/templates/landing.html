title: Suspicious PowerShell Execution
status: experimental

  product: windows
  service: powershell
detection:
  selection:
    CommandLine|contains:
      - '-encodedcommand'
      - '-enc'
      - 'bypass'
      - 'hidden'
  condition: selection
level: high
response:
  action: kill_process
  requires_approval: true
